逐字打亮（为什么选择微光、前端的内容部分）


HTML:<h1 class="glowIn">Hello World</h1>
<p class="glowIn">Hello</p>
css:.glowIn {
color: white;
span {
animation: glow-in 0.8s both;
}
}
@keyframes glow-in {
from {
opacity: 0;
}
65% {
opacity: 1;
text-shadow: 0 0 25px white;
}
75% {
opacity: 1;
}
to {
opacity: 0.7;
}
}
JS:let glowInTexts = document.querySelectorAll(".glowIn");//glowInTexts数组将包含两个元素,一个h1,一个p
glowInTexts.forEach(glowInText => {//第一次glowInTexts[0]被赋值给glowInText,它实际是h1;第二次glowInTexts[1]被赋值给glowInText,它实际是p
let letters = glowInText.textContent.split("");//glowInText.textContent获取glowInText中的文本内容;h1时letters将是 ['H', 'e', 'l', 'l', 'o', ' ', 'W', 'o', 'r', 'l', 'd']
glowInText.textContent = "";//清空glowInText的文本内容
letters.forEach((letter, i) => {//遍历letters中每个元素，i是下标
let span = document.createElement("span");
span.textContent = letter;
span.style.animationDelay = `${i * 0.05}s`;
glowInText.append(span);//将span元素添加到glowInText元素的子元素末尾append可换成appendChild
})//不设置span ,glowInText不是实际的DOM元素（标签），无法添加样式
})
